# Step 1 (once): create env
./1_create_environment.sh

# Step 2 (once): set up infra (stays in scripts dir)
./2_setup_build.sh

# Step 3: activate env (still in scripts dir)
source ~/devel/activate_sirf.sh

# Step 4: configure (still in scripts dir; writes to $HOME/devel/SIRF_builds/conda)
./3_configure_sirf.sh

# Step 5: build (still in scripts dir)
./4_build.sh

# Step 6: use SIRF in current shell
source ~/devel/SIRF_builds/conda/INSTALL/bin/env_sirf.sh

# Step 7 (optional): add to bashrc
echo 'source ~/devel/SIRF_builds/conda/INSTALL/bin/env_sirf.sh' >> ~/.bashrc
